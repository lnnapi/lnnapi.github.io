# SpotBalance

# GetAllBalance(Get all balance of the account)

Interface: /SpotBalance/GetAllBalance
Method: GET
Parameter: None

Response:

[
	{
		"Coin": Coin symbol name(String),
		"Available": Available amount(Float64),
		"Frozen": Frozen amount(Float64)
	},
	....
]

# GetBalance(Get balance of one specified coin symbol)

Interface: /SpotBalance/GetBalance
Method: GET
Parameter:

ParameterTypeExplain
CoinStringSymbol of the coin: eg. BTC

Response:

[
    {
		"Coin": Coin symbol name(String),
		"Available": Available amount(Float64),
		"Frozen": Frozen amount(Float64)
	}
    ...
]
